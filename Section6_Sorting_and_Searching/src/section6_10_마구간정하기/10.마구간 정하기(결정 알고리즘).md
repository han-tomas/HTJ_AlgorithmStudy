	public int count(int[] arr, int distance) {
		int count = 1;
		int point = arr[0];

	public int solution(int n, int c, int[] arr) {
		int answer = 0;
		Arrays.sort(arr);

			}
			else rt = mid - 1;
		}
		return answer;
	}
